Grid Areas를 사용하면 복잡한 레이아웃도 쉽게 만들 수 있습니다.

.internal-card {
  display: grid;
  grid-template-areas:
    "image image"
    "title price"
    "description description"
    "button button";
  grid-template-rows: 150px auto auto auto;
  grid-template-columns: 1fr auto;
  gap: 15px;
}

.internal-card .card-image { grid-area: image; }
.internal-card h3 { grid-area: title; }
.internal-card .price { grid-area: price; }
.internal-card p { grid-area: description; }
.internal-card button { grid-area: button; }
