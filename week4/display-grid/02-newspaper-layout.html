.newspaper {
  display: grid;
  grid-template-columns: 1fr 250px;
  grid-template-rows: auto 1fr auto;
  gap: 20px;
  grid-template-areas:
    "header header"
    "article aside"
    "footer footer";
}

/* 각 영역에 grid-area 지정 */
header { grid-area: header; }
article { grid-area: article; }
aside { grid-area: aside; }
footer { grid-area: footer; }
